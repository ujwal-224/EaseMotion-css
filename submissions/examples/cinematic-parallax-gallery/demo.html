Usage & Integration

To integrate the Cinematic Parallax Image Gallery, copy the HTML markup structure and link the CSS style. Place the Javascript script inside your initialization file to handle tracking coordinate states. Use variables to tune sensitivity dynamically. 

<!-- Cinematic Parallax Grid Gallery -->
<div class="ease-cinematic-grid">
  <div class="ease-cinematic-card-wrapper">
    <div class="ease-cinematic-card">
      <div class="ease-cinematic-badge">Category</div>
      <div class="ease-cinematic-media">
        <img class="ease-cinematic-image" src="your-image.jpg" alt="Preview">
        <div class="ease-cinematic-overlay"></div>
      </div>
      <div class="ease-cinematic-glare"></div>
      <div class="ease-cinematic-content">
        <span class="ease-cinematic-subtitle">Subheading</span>
        <h2 class="ease-cinematic-title">Card Title</h2>
        <p class="ease-cinematic-desc">Description floats inside 3D space.</p>
      </div>
    </div>
  </div>
</div>
document.addEventListener("DOMContentLoaded", () => {
  const cards = document.querySelectorAll('.ease-cinematic-card-wrapper');
  
  cards.forEach(wrapper => {
    const card = wrapper.querySelector('.ease-cinematic-card');
    let frameId = null;

    wrapper.addEventListener('mousemove', (e) => {
      if (frameId) cancelAnimationFrame(frameId);

      frameId = requestAnimationFrame(() => {
        const rect = wrapper.getBoundingClientRect();
        const x = e.clientX - rect.left - rect.width / 2;
        const y = e.clientY - rect.top - rect.height / 2;

        const normX = x / (rect.width / 2);
        const normY = y / (rect.height / 2);

        // Fetch CSS custom variable limits dynamically
        const root = document.documentElement;
        const maxTilt = parseFloat(getComputedStyle(root).getPropertyValue('--ease-cinematic-tilt-max')) || 20;
        const maxParallax = parseFloat(getComputedStyle(root).getPropertyValue('--ease-cinematic-parallax-max')) || 15;

        // Opposite translation (negative multiplier)
        const tiltX = -normY * maxTilt;
        const tiltY = normX * maxTilt;
        const parallaxX = -normX * maxParallax;
        const parallaxY = -normY * maxParallax;

        card.classList.remove('resetting');
        card.style.setProperty('--tilt-x', `${tiltX}deg`);
        card.style.setProperty('--tilt-y', `${tiltY}deg`);
        card.style.setProperty('--parallax-x', `${parallaxX}px`);
        card.style.setProperty('--parallax-y', `${parallaxY}px`);

        // Glare dynamic repositioning
        const glareX = ((e.clientX - rect.left) / rect.width) * 100;
        const glareY = ((e.clientY - rect.top) / rect.height) * 100;
        card.style.setProperty('--glare-x', `${glareX}%`);
        card.style.setProperty('--glare-y', `${glareY}%`);
        card.style.setProperty('--glare-opacity', '1');
      });
    });

    wrapper.addEventListener('mouseleave', () => {
      if (frameId) cancelAnimationFrame(frameId);
      
      card.classList.add('resetting');
      card.style.removeProperty('--tilt-x');
      card.style.removeProperty('--tilt-y');
      card.style.removeProperty('--parallax-x');
      card.style.removeProperty('--parallax-y');
      card.style.removeProperty('--glare-opacity');
    });
  });
});
/* Refer to style.css file for full variables and layout classes */
.ease-cinematic-grid {
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(280px, 1fr));
  gap: 2rem;
  perspective: var(--ease-cinematic-perspective);
}

.ease-cinematic-card {
  transform-style: preserve-3d;
  transform: rotateX(var(--tilt-x, 0deg)) rotateY(var(--tilt-y, 0deg));
  transition: transform 150ms cubic-bezier(0.25, 0.61, 0.355, 1);
}

.ease-cinematic-image {
  transform: scale(1.15) translate(var(--parallax-x, 0px), var(--parallax-y, 0px));
  transition: transform 150ms cubic-bezier(0.25, 0.61, 0.355, 1);
}
